public class GradingStudents {
    // https://www.hackerrank.com/challenges/grading/problem

    public GradingStudents() {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int k = in.nextInt();
        int sum = 0;

        for(int i = 0; i < n; i++){
            int temp = in.nextInt();

            if (i != k) sum += temp;
        }

        int b = in.nextInt();
        int c = sum / 2;
        System.out.println(c== b ? "Bon Appetit" : b-c);
    }
}
